    def initialize(first_bucket_max = 0.010, bucket_count = 30, multiplier = 1.3)
      @buckets = []

      min = 0
      max = first_bucket_max

      (bucket_count - 1).times do
        @buckets << Bucket.new(min, max)
        min = max
        max *= multiplier
      end
      @buckets << Bucket.new(max)
    end
