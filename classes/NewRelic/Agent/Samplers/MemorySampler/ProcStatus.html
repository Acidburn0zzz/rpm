      def get_memory
        File.open(proc_status_file, "r") do |f|
          while !f.eof? 
            if f.readline =~ /RSS:\s*(\d+) kB/i
              return $1.to_f / 1024.0
            end
          end
        end
        raise "Unable to find RSS in #{proc_status_file}"
      end
